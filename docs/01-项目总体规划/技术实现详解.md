```typescript
// src/app/api/user/kyc/submit/route.ts
import { NextRequest } from 'next/server';
import { z } from 'zod';
import { pg } from '@/lib/pg';
import { emitEvent } from '@/lib/event-bus';
import { requireAuth } from '@/lib/auth';
import { withIdempotency } from '@/lib/idempotency';
import { notarize } from '@/lib/blockchain';

const kycSchema = z.object({
  documents: z.array(z.object({
    type: z.enum(['id_card', 'passport', 'driver_license', 'selfie']),
    url: z.string().url(),
    hash: z.string().length(64),
  })),
  realName: z.string().min(2).max(64),
  idNumber: z.string().regex(/^[A-Z0-9]{6,20}$/),
});

export async function POST(req: NextRequest) {
  return withIdempotency(req, async () => {
    const user = await requireAuth(req);
    const body = await req.json();
    const data = kycSchema.parse(body);

    // 1. 同步写 PG
    const kycRecord = await pg.query(
      `INSERT INTO kyc_records (user_id, documents, real_name, id_number, status, created_at)
       VALUES ($1, $2, $3, $4, 'submitted', NOW())
       RETURNING *`,
      [user.id, JSON.stringify(data.documents), data.realName, data.idNumber]
    );

    await pg.query(
      `UPDATE users SET kyc_status = 'submitted', updated_at = NOW() WHERE id = $1`,
      [user.id]
    );

    // 2. 异步发 Kafka（不阻塞响应）
    emitEvent('kyc.submitted', {
      userId: user.id,
      kycRecordId: kycRecord.rows[0].id,
      documents: data.documents,
      did: user.did,
    }).catch(err => logger.error({ err }, 'Kafka emit failed'));

    // 3. 异步存证（不阻塞响应）
    notarize({
      bizType: 'kyc',
      bizId: kycRecord.rows[0].id,
      fileHash: data.documents[0].hash,
      did: user.did,
    }).catch(err => logger.error({ err }, 'Notarize failed'));

    return Response.json({
      kycId: kycRecord.rows[0].id,
      status: 'submitted',
      message: 'KYC 提交成功，审核中',
    });
  });
}
```

```typescript
import { Kafka, Producer } from 'kafkajs';
import { getTraceId } from '@/lib/trace';

const kafka = new Kafka({
  clientId: 'zs-exchange',
  brokers: (process.env.KAFKA_BROKERS || 'kafka:9092').split(','),
  retry: { retries: 5, initialRetryTime: 300 },
});

let producer: Producer | null = null;

async function getProducer(): Promise<Producer> {
  if (!producer) {
    producer = kafka.producer({ idempotent: true, maxInFlightRequests: 1 });
    await producer.connect();
  }
  return producer;
}

export async function emitEvent(topic: string, payload: any) {
  const p = await getProducer();
  await p.send({
    topic,
    messages: [{
      key: payload.userId || payload.bizId || payload.id,
      value: JSON.stringify({
        ...payload,
        emittedAt: Date.now(),
      }),
      headers: {
        'trace-id': getTraceId(),
        'event-type': topic,
        'schema-version': '1.0',
      },
    }],
  });
}
```

```typescript
// src/lib/ws-server.ts
import { WebSocketServer, WebSocket } from 'ws';
import { verifyJWT } from '@/lib/auth';
import { pg } from '@/lib/pg';

const wss = new WebSocketServer({ port: 3001, path: '/api/ws/notifications' });

wss.on('connection', async (ws: WebSocket, req) => {
  // 鉴权
  const token = new URL(req.url!, 'http://localhost').searchParams.get('token');
  const user = await verifyJWT(token!);
  if (!user) {
    ws.close(4001, 'Unauthorized');
    return;
  }

  // 订阅用户专属 channel
  const channel = `user:${user.id}`;
  await pg.listen('notifications', async (payload) => {
    const notif = JSON.parse(payload);
    if (notif.userId === user.id) {
      ws.send(JSON.stringify({
        type: 'event',
        channel,
        data: notif,
        ts: Date.now(),
      }));
    }
  });

  // 心跳
  const heartbeat = setInterval(() => {
    if (ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify({ type: 'ping', ts: Date.now() }));
    }
  }, 30000);

  ws.on('close', () => {
    clearInterval(heartbeat);
    pg.unlisten('notifications');
  });
});
```

### 3.2 凭证预生成 + 异步上链

```typescript
// src/lib/blockchain.ts
import { createHash } from 'crypto';
import { sign } from '@/lib/crypto';
import { pg } from '@/lib/pg';
import { Kafka } from 'kafkajs';

const kafka = new Kafka({ brokers: [process.env.KAFKA_BROKERS!] });
const producer = kafka.producer();

export interface NotarizeRequest {
  bizType: string;
  bizId: string;
  fileHash: string;
  did: string;
  metadata?: Record<string, any>;
}

export interface NotarizeResponse {
  certId: string;
  certHash: string;
  timestamp: number;
  sign: string;
  txHash?: string;
  blockNumber?: number;
  verifyUrl: string;
}

export async function notarize(req: NotarizeRequest): Promise<NotarizeResponse> {
  // 1. 生成 certId
  const certId = `cert-${Date.now()}-${Math.random().toString(36).slice(2, 10)}`;

  // 2. 计算 certHash
  const certPayload = {
    certId,
    bizType: req.bizType,
    bizId: req.bizId,
    fileHash: req.fileHash,
    did: req.did,
    timestamp: Date.now(),
  };
  const certHash = createHash('sha256')
    .update(JSON.stringify(certPayload))
    .digest('hex');

  // 3. 本地签名
  const sig = await sign(certHash, process.env.NOTARY_PRIVATE_KEY!);

  // 4. 写 PG (同步)
  await pg.query(
    `INSERT INTO blockchain_notarizations
       (cert_id, biz_type, biz_id, file_hash, did, cert_hash, sign, status, created_at)
     VALUES ($1, $2, $3, $4, $5, $6, $7, 'pending', NOW())`,
    [certId, req.bizType, req.bizId, req.fileHash, req.did, certHash, sig]
  );

  // 5. 异步上链（不阻塞）
  await producer.send({
    topic: 'notarize.pending',
    messages: [{
      key: certId,
      value: JSON.stringify({ certId, certHash, bizType: req.bizType, bizId: req.bizId }),
    }],
  });

  return {
    certId,
    certHash,
    timestamp: certPayload.timestamp,
    sign: sig,
    verifyUrl: `${process.env.PUBLIC_URL}/verify/${certId}`,
  };
}
```

```python
import asyncio
import aiohttp
from aiokafka import AIOKafkaProducer
from datetime import datetime, timedelta

class DouyinCrawler:
    def __init__(self, config):
        self.client_key = config['client_key']
        self.client_secret = config['client_secret']
        self.access_token = None
        self.token_expires_at = None
        self.kafka_producer = AIOKafkaProducer(bootstrap_servers='kafka:9092')

    async def get_access_token(self):
        if self.access_token and datetime.now() < self.token_expires_at:
            return self.access_token
        async with aiohttp.ClientSession() as session:
            async with session.post('https://open.douyin.com/oauth/access_token', json={
                'client_key': self.client_key,
                'client_secret': self.client_secret,
                'grant_type': 'client_credential',
            }) as resp:
                data = await resp.json()
                self.access_token = data['access_token']
                self.token_expires_at = datetime.now() + timedelta(seconds=data['expires_in'] - 300)
                return self.access_token

    async def fetch_videos(self, page=0):
        token = await self.get_access_token()
        async with aiohttp.ClientSession() as session:
            async with session.get(
                'https://open.douyin.com/discovery/search/item/',

                    'access_token': token,
                    'keyword': '股票 投资',
                    'count': 50,
                    'cursor': page * 50,
                }
            ) as resp:
                data = await resp.json()
```

## 六、关键技术决策

### 6.1 决策 1：数据库共享 vs 隔离

| 方案 | 优势 | 劣势 | 决策 |
|------|------|------|------|
| 共享 PG (单一库) | 简单 / JOIN 方便 / 事务 | 表冲突 | ❌ |
| 共享 PG（多 schema）| 隔离 / JOIN 方便 | 命名冲突 | ✅ |
| 独立 DB | 完全隔离 | 跨库 JOIN 复杂 | ❌ |

**最终决策**：共享 PostgreSQL，引擎表用独立 schema（`n8n.`, `flowable.`, `fabric.`）

### 6.2 决策 2：API 代理 vs 直连

| 方案 | 优势 | 劣势 | 决策 |
|------|------|------|------|
| 前端直连引擎 | 低延迟 | CORS / 安全 | ❌ |
| 统一代理 | 鉴权 / 限流 / 缓存 | 多一跳 | ✅ |

**最终决策**：所有外部引擎通过 API Gateway 代理，前端只与 ZS Exchange 通信

### 6.3 决策 3：同步 vs 异步

| 场景 | 决策 | 原因 |
|------|------|------|
| 登录 | 同步 | 用户感知 |
| 存证凭证 | 同步 (本地签名) + 异步 (上链) | 凭证必须立即返回 |
| AI 风险评估 | 异步 | 复杂分析 |
| WS 推送 | 异步 | 实时性 |

### 6.4 决策 4：存证一致性

| 方案 | 优势 | 劣势 | 决策 |
|------|------|------|------|
| 同步上链 | 强一致 | 慢 (5-10s) | ❌ |
| 异步上链 + 立即凭证 | 快速 + 可验 | 最终一致 | ✅ |
| 不上链 | 最快 | 无法律效力 | ❌ |

**最终决策**：本地立即生成凭证 + 异步上链 + 二维码验证

### 8.1 鉴权

```typescript
// 多层鉴权
// 1. JWT 验证
// 2. RBAC 权限
// 3. Scope 校验
// 4. 资源所有权
// 5. 速率限制
// 6. IP 黑白名单
// 7. 设备指纹

import { jwtVerify } from 'jose';

export async function authMiddleware(req: NextRequest) {
  // 1. 提取 token
  const token = req.headers.get('authorization')?.replace('Bearer ', '');
  if (!token) throw new AuthError(401, 'Missing token');

  // 2. 验证签名
  const { payload } = await jwtVerify(
    token,
    new TextEncoder().encode(process.env.JWT_SECRET!),
    { algorithms: ['HS256'] },
  );

  // 3. 验证过期
  if (payload.exp! < Date.now() / 1000) {
    throw new AuthError(401, 'Token expired');
  }

  // 4. 黑名单检查
  const isRevoked = await redis.get(`jwt:revoked:${token}`);
  if (isRevoked) throw new AuthError(401, 'Token revoked');

  // 5. 附加 user / scope
  req.user = payload;
  req.scopes = (payload.scopes as string[]) || [];
}
```
