### 3.3 心跳与重连

```typescript
// 心跳：30s

// 服务端响应
{ type: 'pong', ts: 1718000000000 }

// 客户端重连策略
const reconnect = (url) => {
  let attempts = 0;
  const max = 10;
  const connect = () => {
    const ws = new WebSocket(url);
    ws.onclose = () => {
      attempts++;
      if (attempts < max) {
        setTimeout(connect, Math.min(1000 * 2 ** attempts, 30000));
      }
    };
  };
  connect();
};
```

### 6.2 集成抽象层

```typescript
// src/lib/integrations/llm/index.ts
export interface LLMProvider {
  chat(prompt: string, options?: any): Promise<ChatResponse>;
  embed(text: string): Promise<number[]>;
  streamChat(prompt: string): AsyncIterable<string>;
}

export class LLMFactory {
  static create(provider: string): LLMProvider {
    switch (provider) {
      case 'openai': return new OpenAIProvider();
      case 'anthropic': return new AnthropicProvider();
      case 'zhipu': return new ZhipuProvider();
      case 'tongyi': return new TongyiProvider();
      default: throw new Error(`Unknown provider: ${provider}`);
    }
  }
}

// 使用
const llm = LLMFactory.create(modelConfig.provider);
const response = await llm.chat(prompt);

// 自动降级
async function chatWithFallback(prompt: string): Promise<string> {
  const providers = ['openai', 'anthropic', 'tongyi'];
  for (const p of providers) {
    try {
      return await LLMFactory.create(p).chat(prompt);
    } catch (e) {
      logger.warn({ provider: p, error: e }, 'LLM failed, trying next');
    }
  }
  throw new Error('All LLM providers failed');
}
```

## 十、安全规范

### 10.1 鉴权

```typescript
// JWT + RBAC
// 所有 API 必须经过鉴权中间件
// 特殊接口需要 scope 校验
export const requireScope = (scope: string) => {
  return async (req, res, next) => {
    const token = req.headers.authorization?.replace('Bearer ', '');
    const payload = await verifyJWT(token);
    if (!payload.scopes?.includes(scope)) {
      return res.status(403).json({ code: 403, message: 'Insufficient scope' });
    }
    next();
  };
};
```

## 十二、监控与可观测性

### 12.1 必埋点

```typescript
// TraceID 全链路
// Metrics 上报 Prometheus
// Logs 上报 Loki

import { trace, context } from '@opentelemetry/api';
import { metrics } from '@opentelemetry/api';

const tracer = trace.getTracer('zs-api');
const meter = metrics.getMeter('zs-api');
const requestCounter = meter.createCounter('api_requests_total');
const latencyHistogram = meter.createHistogram('api_duration_ms');

export async function handleRequest(req: Request) {
  const span = tracer.startSpan(`${req.method} ${req.url}`);
  const start = Date.now();
  try {
    const result = await doBusiness(req);
    span.setStatus({ code: SpanStatusCode.OK });
    return result;
  } catch (e) {
    span.recordException(e);
    span.setStatus({ code: SpanStatusCode.ERROR });
    throw e;
  } finally {
    const duration = Date.now() - start;
    requestCounter.add(1, { path: req.url, method: req.method, status: 200 });
    latencyHistogram.record(duration, { path: req.url });
    span.end();
  }
}
```
