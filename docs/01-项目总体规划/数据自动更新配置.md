```javascript
  },

});

// 速率限制处理
jiraClient.interceptors.response.use(
  response => response,
  async error => {
    if (error.response?.status === 429) {
      const retryAfter = error.response.headers['retry-after'] || 60;
      console.log(`⏳ Jira 限流，${retryAfter}s 后重试`);
      await new Promise(r => setTimeout(r, retryAfter * 1000));
      return jiraClient.request(error.config);
    }
    return Promise.reject(error);
  }
);

module.exports = jiraClient;
```

```javascript
// jira-sync-tasks.js
async function syncTasks() {
  const jql = `
    project = ZS-AI 
    AND issuetype in (Story, Task, Bug, Epic) 
    AND status != Closed
    ORDER BY priority DESC, updated DESC
  `;
  
  let allIssues = [];
  let startAt = 0;
  const maxResults = 100;
  
  // 分页拉取
  while (true) {
    const response = await jiraClient.post('/search', {
      jql,
      startAt,
      maxResults,
      fields: [
        'summary', 'status', 'priority', 'assignee', 'reporter',
        'created', 'updated', 'duedate', 'resolutiondate',
        'timeoriginalestimate', 'timespent', 'progress',
        'customfield_10100', // 阶段
        'customfield_10101', // Sprint
        'customfield_10102', // 故事点
        'sprint'
      ],
      expand: ['sprint', 'names']
    });
    
    allIssues = allIssues.concat(response.data.issues);
    if (startAt + maxResults >= response.data.total) break;
    startAt += maxResults;
  }
  
  return transformTasks(allIssues);
}

function transformTasks(issues) {
  return issues.map(issue => ({
    jiraId: issue.key,
    title: issue.fields.summary,
    type: issue.fields.issuetype.name,
    status: issue.fields.status.name,
    priority: issue.fields.priority.name,
    assignee: issue.fields.assignee?.displayName,
    storyPoints: issue.fields.customfield_10102,
    sprint: extractSprintName(issue.fields.sprint),
    stage: issue.fields.customfield_10100,
    created: issue.fields.created,
    updated: issue.fields.updated,
    dueDate: issue.fields.duedate,
    progress: issue.fields.progress?.percent || 0
  }));
}
```

### 2.4 Jira Webhook 实时推送

```javascript
// jira-webhook-handler.js
const express = require('express');
const app = express();

app.post('/webhook/jira', express.json(), async (req, res) => {
  const webhookEvent = req.body;
  
  // 验证 Webhook 来源
  if (!verifyJiraSignature(req)) {
    return res.status(401).send('Unauthorized');
  }
  
  switch (webhookEvent.webhookEvent) {
    case 'jira:issue_created':
      await handleIssueCreated(webhookEvent.issue);
      break;
    case 'jira:issue_updated':
      await handleIssueUpdated(webhookEvent.issue);
      break;
    case 'jira:issue_deleted':
      await handleIssueDeleted(webhookEvent.issue);
      break;
    case 'sprint_started':
      await handleSprintStarted(webhookEvent.sprint);
      break;
    case 'sprint_closed':
      await handleSprintClosed(webhookEvent.sprint);
      break;
  }
  
  res.status(200).send('OK');
});

async function handleIssueUpdated(issue) {
  // 1. 同步到 Notion
  await notionSync.updateTask(issue);
  // 2. 同步到 Confluence
  await confluenceSync.updateTask(issue);
  // 3. 更新 ClickHouse
  await clickhouse.insert('tasks', transformIssue(issue));
  // 4. 检查是否触发风险告警
  await checkRiskTriggers(issue);
}
```

### 5.2 签名校验与发送

```javascript
// feishu-webhook.js
const crypto = require('crypto');

class FeishuWebhook {
  constructor(webhookUrl, secret = null) {
    this.webhookUrl = webhookUrl;
    this.secret = secret;
  }
  
  // 计算签名
  generateSign(timestamp) {
    if (!this.secret) return '';
    const stringToSign = `${timestamp}\n${this.secret}`;
    const hmac = crypto.createHmac('sha256', stringToSign);
    return hmac.update('').digest('base64');
  }
  
  // 发送文本消息
  async sendText(content, atMobiles = []) {
    const timestamp = Math.floor(Date.now() / 1000);
    const sign = this.generateSign(timestamp);
    
    return this.send({
      timestamp,
      sign,
      msg_type: 'text',
      content: {
        text: content + (atMobiles.length 
          ? '\n@' + atMobiles.join(' @') 
          : '')
      }
    });
  }
  
  // 发送卡片消息
  async sendCard(card) {
    const timestamp = Math.floor(Date.now() / 1000);
    const sign = this.generateSign(timestamp);
    
    return this.send({
      timestamp,
      sign,
      msg_type: 'interactive',
      card
    });
  }
  
  async send(payload) {
    const response = await fetch(this.webhookUrl, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    });
    return response.json();
  }
  
  // 发送 P0 告警
  async sendP0Alert(risk) {
    return this.sendCard({
      header: {
        title: {
          tag: 'plain_text',
          content: `🚨 P0 告警 · ${risk.title}`
        },
        template: 'red'
      },
      elements: [
        {
          tag: 'div',
          text: {
            tag: 'lark_md',
            content: this.formatRiskContent(risk)
          }
        },
        {
          tag: 'action',
          actions: [
            {
              tag: 'button',
              text: { tag: 'plain_text', content: '查看 Jira' },
              url: risk.jiraUrl,
              type: 'primary'
            },
            {
              tag: 'button',
              text: { tag: 'plain_text', content: '运行手册' },
              url: risk.runbookUrl,
              type: 'default'
            }
          ]
        }
      ]
    });
  }
  
  formatRiskContent(risk) {
    return [
      `**风险 ID**：${risk.id}`,
      `**等级**：🔴 ${risk.level}`,
      `**影响范围**：${risk.impact}`,
      `**触发时间**：${risk.triggeredAt}`,
      `**负责人**：<at id="${risk.assigneeId}">${risk.assignee}</at>`,
      `**状态**：${risk.status}`,
      '',
      '**📋 应急步骤**：',
      ...risk.steps.map((s, i) => `${i + 1}. ${s}`)
    ].join('\n');
  }
}

module.exports = FeishuWebhook;
```

```python
import requests
import pandas as pd
from typing import List, Dict

class JiraExtractor:
    def __init__(self, base_url, auth):
        self.base_url = base_url
        self.auth = auth
    
    def extract_tasks(self, jql: str, batch_size: int = 100) -> List[Dict]:
        all_issues = []
        start_at = 0
        
        while True:
            response = requests.post(
                f'{self.base_url}/search',
                auth=self.auth,
                json={
                    'jql': jql,
                    'startAt': start_at,
                    'maxResults': batch_size,
                    'fields': ['*all']
                }
            )
            response.raise_for_status()
            data = response.json()
            
            all_issues.extend(data['issues'])
            
            if start_at + batch_size >= data['total']:
                break
            start_at += batch_size
        
        return all_issues

class GitLabExtractor:
    def __init__(self, base_url, token):
        self.base_url = base_url
        self.headers = {'PRIVATE-TOKEN': token}
    
    def extract_mr_stats(self, project_id: str, since: str) -> List[Dict]:
        response = requests.get(
            f'{self.base_url}/api/v4/projects/{project_id}/merge_requests',
            headers=self.headers,
            params={
                'created_after': since,
                'per_page': 100,
                'state': 'all'
            }
        )
        return response.json()

class PrometheusExtractor:
    def __init__(self, base_url):
        self.base_url = base_url
    
    def extract_metrics(self, query: str, time_range: str = '1h') -> Dict:
        response = requests.get(
            f'{self.base_url}/api/v1/query_range',
            params={'query': query, 'range': time_range}
```

```python
# etl/transform.py
import pandas as pd
from datetime import datetime

def transform_tasks(raw_issues: list) -> pd.DataFrame:
    records = []
    for issue in raw_issues:
        f = issue['fields']
        records.append({
            'jira_id': issue['key'],
            'title': f['summary'],
            'type': f['issuetype']['name'],
            'status': f['status']['name'],
            'priority': f['priority']['name'],
            'assignee': f['assignee']['displayName'] if f['assignee'] else None,
            'stage': f.get('customfield_10100', {}).get('value'),
            'sprint': extract_sprint_name(f.get('sprint')),
            'story_points': f.get('customfield_10102'),
            'estimated_hours': f.get('timeoriginalestimate', 0) / 3600,
            'spent_hours': f.get('timespent', 0) / 3600,
            'progress_percent': f.get('progress', {}).get('percent', 0),
            'created_at': parse_date(f['created']),
            'updated_at': parse_date(f['updated']),
            'due_date': parse_date(f.get('duedate')),
            'synced_at': datetime.now()
        })
    
    df = pd.DataFrame(records)
    
    # 数据质量处理
    df['is_overdue'] = (
        (df['status'] != 'Done') & 
        (df['due_date'] < datetime.now())
    )
    df['burnout_risk'] = (
        df['spent_hours'] > df['estimated_hours'] * 1.2
    )
    
    return df

def transform_risks(raw_risks: list) -> pd.DataFrame:
    records = []
    for risk in raw_risks:
        f = risk['fields']
        priority = f['priority']['name']
        impact = f.get('customfield_10200', {}).get('value', '中')
        probability = f.get('customfield_10201', {}).get('value', '中')
        
        records.append({
            'risk_id': risk['key'],
            'title': f['summary'],
            'level': priority,
            'category': f.get('customfield_10203', {}).get('value'),
            'impact': impact,
            'probability': probability,
            'score': calculate_risk_score(priority, impact, probability),
            'assignee': f['assignee']['displayName'] if f['assignee'] else None,
            'status': f['status']['name'],
            'created_at': parse_date(f['created']),
            'synced_at': datetime.now()
        })
    
    return pd.DataFrame(records)

def calculate_risk_score(priority, impact, probability):
    weights = {'P0': 100, 'P1': 50, 'P2': 20, 'P3': 5}
    impact_coef = {'高': 3, '中': 2, '低': 1}
    prob_coef = {'高': 3, '中': 2, '低': 1}
    return (
        weights.get(priority, 10) * 
        impact_coef.get(impact, 1) * 
        prob_coef.get(probability, 1)
    )
```

```python
# etl/load.py
from clickhouse_driver import Client
import pandas as pd

class ClickHouseLoader:
    def __init__(self, host, port, database, user, password):
        self.client = Client(
            host=host, port=port, database=database,
            user=user, password=password
        )
    
    def load_dataframe(self, table: str, df: pd.DataFrame):
        # 转换为 records
        data = df.to_dict('records')
        # 批量插入
        self.client.execute(
            f'INSERT INTO {table} VALUES',
            data
        )
    
    def upsert(self, table: str, df: pd.DataFrame, key_col: str):
        """使用 ReplacingMergeTree 实现 upsert"""
        # 1. 删除旧数据
        keys = df[key_col].tolist()
        self.client.execute(
            f'ALTER TABLE {table} DELETE WHERE {key_col} IN ({",".join(keys)})'
        )
        # 2. 插入新数据
        self.load_dataframe(table, df)
```

### 8.1 Notion 同步客户端

```javascript
// notion-sync.js
const { Client } = require('@notionhq/client');

class NotionSync {
  constructor(token) {
    this.notion = new Client({ auth: token });
  }
  
  // 批量更新任务
  async updateTasks(databaseId, tasks) {
    for (const task of tasks) {
      await this.upsertTask(databaseId, task);
      // 避免 API 限流
      await this.sleep(350);
    }
  }
  
  async upsertTask(databaseId, task) {
    // 1. 查询现有页面
    const existing = await this.notion.databases.query({
      database_id: databaseId,
      filter: {
        property: 'Jira ID',
        title: { equals: task.jiraId }
      }
    });
    
    const properties = this.buildTaskProperties(task);
    
    if (existing.results.length > 0) {
      // 更新现有页面
      await this.notion.pages.update({
        page_id: existing.results[0].id,
        properties
      });
    } else {
      // 创建新页面
      await this.notion.pages.create({
        parent: { database_id: databaseId },
        properties,
        children: this.buildTaskBlocks(task)
      });
    }
  }
  
  buildTaskProperties(task) {
    return {
      '任务标题': {
        title: [{ text: { content: task.title } }]
      },
      'Jira ID': {
        rich_text: [{ text: { content: task.jiraId } }]
      },
      '状态': {
        status: { name: this.mapStatus(task.status) }
      },
      '优先级': {
        select: { name: task.priority }
      },
      '负责人': {
        people: task.notionUserId 
          ? [{ id: task.notionUserId }] 
          : []
      },
      '完成度': {
        number: task.progress
      },
      '开始日期': task.startDate ? {
        date: { start: task.startDate }
      } : undefined,
      '截止日期': task.dueDate ? {
        date: { start: task.dueDate }
      } : undefined
    };
  }
  
  mapStatus(jiraStatus) {
    const map = {
      'To Do': 'To-do',
      'In Progress': 'In Progress',
      'In Review': 'In Review',
      'Done': 'Done',
      'Blocked': 'Blocked'
    };
    return map[jiraStatus] || jiraStatus;
  }
  
  sleep(ms) {
    return new Promise(r => setTimeout(r, ms));
  }
}

module.exports = NotionSync;
```

### 示例 1：每日数据同步全流程

**场景**：每 15 分钟自动同步 Jira 任务数据到 Notion 和 Confluence

**步骤**：

1. **Cron 触发** → 14:00 `sync-jira-tasks.sh` 启动
2. **Jira 拉取** → 通过 REST API 拉取近 1 小时变更的任务
3. **数据转换** → 标准化为 Notion/Confluence 字段
4. **数据验证** → 检查必填字段、值域、唯一性
5. **加载到 ClickHouse** → 持久化存储
6. **同步到 Notion** → 通过 Notion API 更新/创建页面
7. **同步到 Confluence** → 更新 Jira Issue Macro 渲染
8. **飞书通知** → 发送"数据同步完成"消息
9. **Prometheus 上报** → 记录 `zs_ai_sync_duration_seconds`
10. **审计日志** → 记录同步数量、状态、耗时

**预期结果**：
- 仪表板数据延迟 ≤ 5 分钟
- 同步成功率 ≥ 99%
- 任何失败自动告警

### 示例 2：P0 风险实时告警全流程

**场景**：智能合约出现 Critical 漏洞，触发 P0 告警

**步骤**：

1. **Prometheus 抓取** → `contract_vuln_critical_count > 0`
2. **Alertmanager 接收** → 评估规则，匹配 P0 路由
3. **抑制规则** → 抑制同源的 P1/P2 告警
4. **多渠道通知** → 
   - 飞书群（带卡片 + 按钮）
   - 邮件（含详情）
   - 短信（On-Call）
   - 电话（PagerDuty）
5. **自动升级** → 5 分钟未响应 → 升级到 L2
6. **Jira 联动** → 自动创建 P0 风险登记
7. **状态同步** → Notion/Confluence 仪表板更新
8. **飞书 On-Call** → 强制确认接收
9. **事件记录** → Sentry/审计日志
10. **关闭流程** → 修复后复盘，归档

**预期结果**：
- 5 分钟内所有相关人员收到通知
- 升级路径明确，无遗漏
- 完整事件链可追溯
