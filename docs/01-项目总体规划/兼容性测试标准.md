### 4.3 WebSocket 测试

```typescript
// tests/contract/ws.test.ts
describe('WebSocket Compatibility', () => {
  it('should push n8n execution status', async () => {
    const ws = new WebSocket('wss://api.zs.exchange/api/ws/n8n?token=JWT');
    const messages: any[] = [];

    await new Promise<void>((resolve, reject) => {
      const timeout = setTimeout(() => reject(new Error('Timeout')), 10000);
      ws.onopen = () => {
        ws.send(JSON.stringify({ type: 'subscribe', channel: 'execution.status', filter: { workflowId: 'wf-001' } }));
      };
      ws.onmessage = (e) => {
        const msg = JSON.parse(e.data);
        messages.push(msg);
        if (msg.type === 'event' && msg.data.status === 'completed') {
          clearTimeout(timeout);
          resolve();
        }
      };
      ws.onerror = reject;

      // 触发工作流执行
      setTimeout(() => fetch('https://api.zs.exchange/api/n8n/proxy/workflows/wf-001/execute', { method: 'POST' }), 1000);
    });

    expect(messages.some(m => m.type === 'event')).toBe(true);
    ws.close();
  });
});
```

### 5.2 压测脚本

```javascript
// tests/load/api-50agents.js
import http from 'k6/http';
import { check, sleep } from 'k6';
import { Trend, Rate } from 'k6/metrics';

const agentLatency = new Trend('agent_latency');
const errorRate = new Rate('errors');

export const options = {
  stages: [
    { duration: '1m', target: 10 },
    { duration: '3m', target: 50 },
    { duration: '10m', target: 50 },
    { duration: '1m', target: 100 },  // 峰值
    { duration: '1m', target: 0 },
  ],
  thresholds: {
    http_req_duration: ['p(95)<3000', 'p(99)<5000'],
    http_req_failed: ['rate<0.01'],
    agent_latency: ['p(95)<3000'],
  },
};

export default function () {
  const res = http.post(
    'https://api.zs.exchange/api/ai/chat',
    JSON.stringify({
      agentId: `agent-${__VU}`,
      message: '执行一个测试任务',
    }),
    {
      headers: {
        'Content-Type': 'application/json',
        Authorization: `Bearer ${__ENV.TOKEN}`,
      },
    }
  );

  agentLatency.add(res.timings.duration);
  errorRate.add(res.status !== 200);

  check(res, {
    'is status 200': (r) => r.status === 200,
    'has response': (r) => r.json('content') !== '',
    'latency < 3s': (r) => r.timings.duration < 3000,
  });

  sleep(1);
}
```
