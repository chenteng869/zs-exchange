# CodeGraph MCP 配置文件
#
# ⚠️ 重要：.vscode/mcp.json 文件在某些 IDE 中是只读的（受保护），
# 所以我们提供 3 种 IDE 的配置文件示例，放在 docs/03-开发工具/ 下。
# 复制对应配置文件到 IDE 的 MCP 配置位置即可启用。

# ============================================================
# 方案 1: VSCode / Cursor / Trae IDE（项目级）
# ============================================================
# 文件: .vscode/mcp.json （如不可写，复制以下内容）
{
  "servers": {
    "codegraph": {
      "type": "stdio",
      "command": "codegraph",
      "args": [
        "serve",
        "--mcp",
        "--path",
        "${workspaceFolder}"
      ],
      "description": "CodeGraph - 自动识别项目根目录"
    }
  }
}
