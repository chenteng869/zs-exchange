## 第一批基础代码：Decimal



`libs/decimal/src/decimal.util.ts`



```TypeScript
import Decimal from 'decimal.js';

export class DecimalUtil {
  static toDecimal(value: string | number | Decimal): Decimal {
    return new Decimal(value);
  }

  static add(a: string | Decimal, b: string | Decimal): Decimal {
    return new Decimal(a).add(new Decimal(b));
  }

  static sub(a: string | Decimal, b: string | Decimal): Decimal {
    return new Decimal(a).sub(new Decimal(b));
  }

  static mul(a: string | Decimal, b: string | Decimal): Decimal {
    return new Decimal(a).mul(new Decimal(b));
  }

  static div(a: string | Decimal, b: string | Decimal): Decimal {
    return new Decimal(a).div(new Decimal(b));
  }

  static min(a: string | Decimal, b: string | Decimal): Decimal {
    return Decimal.min(new Decimal(a), new Decimal(b));
  }

  static max(a: string | Decimal, b: string | Decimal): Decimal {
    return Decimal.max(new Decimal(a), new Decimal(b));
  }

  static toFixed18(value: string | Decimal): string {
    return new Decimal(value).toFixed(18);
  }
}
```

## 第一批核心业务：369 分账计算



`libs/common/src/revenue/wine-369-revenue.ts`



```TypeScript
import { DecimalUtil } from '@fj369/decimal';

export interface RevenueAllocationItemResult {
  pool_type: string;
  percentage: string;
  amount: string;
}

export interface Wine369RevenueResult {
  paid_amount: string;
  tax_amount: string;
  net_amount: string;
  currency: string;
  rule_version: string;
  items: RevenueAllocationItemResult[];
}

export function calculateWine369Revenue(params: {
  paid_amount: string;
  tax_amount?: string;
  currency: string;
  rule_version: string;
}): Wine369RevenueResult {
  const taxAmount = params.tax_amount ?? '0';
  const netAmount = DecimalUtil.sub(params.paid_amount, taxAmount);

  return {
    paid_amount: DecimalUtil.toFixed18(params.paid_amount),
    tax_amount: DecimalUtil.toFixed18(taxAmount),
    net_amount: DecimalUtil.toFixed18(netAmount),
    currency: params.currency,
    rule_version: params.rule_version,
    items: [
      {
        pool_type: 'wine_cost_pool',
        percentage: '0.400000000000000000',
        amount: DecimalUtil.toFixed18(DecimalUtil.mul(netAmount, '0.40'))
      },
      {
        pool_type: 'market_ecosystem_pool',
        percentage: '0.300000000000000000',
        amount: DecimalUtil.toFixed18(DecimalUtil.mul(netAmount, '0.30'))
      },
      {
        pool_type: 'company_pool',
        percentage: '0.300000000000000000',
        amount: DecimalUtil.toFixed18(DecimalUtil.mul(netAmount, '0.30'))
      }
    ]
  };
}
```
